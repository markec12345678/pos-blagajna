package config

// CorsConfig definira CORS dovoljene izvore, metode in headerje.
// Uporablja se v middlewares.AllowCorsWithConfig().
type CorsConfig struct {
	AllowedOrigins   []string `mapstructure:"allowed_origins" yaml:"allowed_origins"`
	AllowedMethods   []string `mapstructure:"allowed_methods" yaml:"allowed_methods"`
	AllowedHeaders   []string `mapstructure:"allowed_headers" yaml:"allowed_headers"`
	AllowCredentials bool     `mapstructure:"allow_credentials" yaml:"allow_credentials"`
	MaxAgeSeconds    int      `mapstructure:"max_age_seconds" yaml:"max_age_seconds"`
}

// RateLimitConfig drži nastavitve za omejevanje hitrosti na občutljivih
// endpointih (login, register, password reset).
type RateLimitConfig struct {
	Enabled            bool `mapstructure:"enabled" yaml:"enabled"`
	LoginMaxRequests   int  `mapstructure:"login_max_requests" yaml:"login_max_requests"`
	LoginWindowSeconds int  `mapstructure:"login_window_seconds" yaml:"login_window_seconds"`
	LoginMaxFailures   int  `mapstructure:"login_max_failures" yaml:"login_max_failures"`
	LoginBlockMinutes  int  `mapstructure:"login_block_minutes" yaml:"login_block_minutes"`
}
